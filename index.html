import csv import random import json import os # 🏫 अपने स्कूल का नाम यहाँ बदलें SCHOOL_NAME = "राजकीय उच्च माध्यमिक विद्यालय,मानेवाला,सूरतगढ़" def load_subject_csv(filename, default_subject_name): questions_pool = [] # अगर फाइल नहीं है, तो टीचर की मदद के लिए एक सैंपल फाइल ऑटोमैटिक बना दी जाएगी if not os.path.exists(filename): with open(filename, mode="w", encoding="utf-8", newline="") as file: writer = csv.writer(file) writer.writerow(["Question", "Opt1", "Opt2", "Opt3", "Opt4", "Answer"]) writer.writerow([f"{default_subject_name} का सैंपल प्रश्न 1?", "विकल्प A", "विकल्प B", "विकल्प C", "विकल्प D", "विकल्प A"]) writer.writerow([f"{default_subject_name} का सैंपल प्रश्न 2?", "उत्तर X", "उत्तर Y", "उत्तर Z", "उत्तर W", "उत्तर X"]) print(f"📝 आपके लिए '{filename}' का ढांचा तैयार कर दिया गया है। कृपया इसमें अपने प्रश्न बदल लें।") # फाइल से प्रश्न पढ़ना with open(filename, mode="r", encoding="utf-8") as file: reader = csv.DictReader(file) for row in reader: if row.get("Question"): # खाली लाइनों को छोड़ने के लिए questions_pool.append({ "question": row["Question"], "options": [row["Opt1"], row["Opt2"], row["Opt3"], row["Opt4"]], "answer": row["Answer"] }) return questions_pool def build_mega_portal(): # तीनों CSV फाइलों से प्रश्न लोड करना math_pool = load_subject_csv("math_questions.csv", "गणित") hindi_pool = load_subject_csv("hindi_questions.csv", "हिंदी") science_pool = load_subject_csv("science_questions.csv", "विज्ञान") # हर विषय से 10 रैंडम प्रश्न चुनना (सुरक्षा जांच के साथ) master_quiz_data = { "math": random.sample(math_pool, min(len(math_pool), 10)), "hindi": random.sample(hindi_pool, min(len(hindi_pool), 10)), "science": random.sample(science_pool, min(len(science_pool), 10)) } master_json = json.dumps(master_quiz_data, ensure_ascii=False) # 🎛️ मास्टर HTML टेम्पलेट (एक ही साइट पर सब कुछ) html_template = f""" मल्टी-सब्जेक्ट डिजिटल परीक्षा पोर्टल

🏫 {SCHOOL_NAME}

🎯 डिजिटल एकीकृत मुख्य परीक्षा पोर्टल
⏱️ शेष समय: 20:00 मिनट

📋 परीक्षा फॉर्म (Exam Registration)

📜 टीचर डैशबोर्ड (एकीकृत परीक्षा परिणाम)

नाम रोल नंबर विषय तारीख प्राप्तांक ग्रेड
""" with open("mega_school_portal.html", "w", encoding="utf-8") as file: file.write(html_template) print("\n🎯 कमाल हो गया! 'mega_school_portal.html' वेबसाइट सफलतापूर्वक बन गई है।") print("👉 इसे ओपन करें, छात्र अपना विषय चुनकर लाइव परीक्षा दे सकते हैं।") if __name__ == "__main__": build_mega_portal()
